Plotting female rappers

rankings %>% 
  filter(gender == "female") %>% 
  ggplot() +
  geom_text(
    aes(
      y = points, 
      x = year,
      label = fontawesome("fa-microphone")),
    colour = "#5C4F41",
    family='fontawesome-webfont'
  ) +
  geom_label_repel(
    aes(
      y = points, 
      x = year, 
      label = artist_song,
      fill = artist_song
    ), 
    data = label_data,
    label.padding = unit(0.2, "lines"), 
    label.size = 0,
    color = "black",
    min.segment.length = 0,
    segment.color = "#5C4F41",
    family = "zcool"
    # arrow = arrow(angle = 30, length = unit(0.1, "inches")),
    # nudge_x = 5,
    # nudge_y = 1
  ) +
  scale_fill_manual(
    values = c(
      "#ccd5ae",
      "#e9edc9",
      "#e6dca5",
      "#faedcd",
      "#d4a373",
      "#c6c98d",
      "#e6d4ba"
    )
  ) +
  labs(
    title = "top songs by female rappers.",
    x = "Year",
    y = "Points",
    caption = "Data Source: BBC Music"
  ) +
  theme(
    legend.position = "none",
    axis.text = element_text(family = "rubik"),
    plot.title = element_text(family = "rubik", 
                              face = "bold",
                              size = 20,
                              hjust = 1,
                              vjust = 1),
    axis.title = element_text(family = "rubik")
    
  ) +
  coord_fixed(ratio = 1/1.5)
